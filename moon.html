# Install (Python 3.9+)

pip install flask skyfield pytz

# app.py

from flask import Flask, request, jsonify
from skyfield.api import load, wgs84
from skyfield import almanac
import pytz, datetime

app = Flask(**name**)
ts = load.timescale()
eph = load(‘de421.bsp’)

@app.route(’/moon’)
def moon():
iso = request.args.get(‘datetime’)   # e.g. 2025-04-10T12:00:00
tz  = request.args.get(‘tz’, ‘UTC’)
zone = pytz.timezone(tz)
dt_local = datetime.datetime.fromisoformat(iso)
dt_utc = zone.localize(dt_local).astimezone(pytz.utc)
t = ts.from_datetime(dt_utc)
earth, moon, sun = eph[‘earth’], eph[‘moon’], eph[‘sun’]
m = (earth + wgs84.latlon(0,0)).at(t).observe(moon).apparent()
s = (earth + wgs84.latlon(0,0)).at(t).observe(sun).apparent()
ml, _, _ = m.ecliptic_latlon()
sl, _, _ = s.ecliptic_latlon()
elong = (ml.degrees - sl.degrees) % 360
age = elong / 360 * 29.530588861
return jsonify({“age”: age, “elongation”: elong,
“moon_lon”: ml.degrees, “sun_lon”: sl.degrees})

if **name** == ‘**main**’: app.run(debug=True)
2
